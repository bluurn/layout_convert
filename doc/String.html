def latinish?
  latin_count = cyrillic_count = 0
  split_words.each do |word|
    if word.latin?
      latin_count+=1
    else
      cyrillic_count+=1
    end
  end
  latin_count > cyrillic_count
end

def swap_layout
  if self.latinish? || self.cyrillish?
    split_words.map do |word|
      layout_map = set_layout word
      word.scan(/./).map do |ch|
        layout_map[ch].nil? ? ch : layout_map[ch]
      end.join
    end.join ' '
  end
end
